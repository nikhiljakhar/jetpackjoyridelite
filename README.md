# How to open
1. Download or clone this repository.
2. cd /path/to/this/repository
3. cd build
4. ./graphics_asgn1
 
# CONTROLS
LEFT KEY = move left
RIGHT KEY = move right
UP KEY = jump
SPACE = jetpack
KEY B = throw balloon
scroll up = zoom in
scroll down = zoom out
left mouse click = return screen to orignal size
